## Dataset Loading with Video-Disjoint Splits

> **Critical:** Always use `split_manifest.json` — not the `train/valid/test` folders
> generated by Roboflow. The Roboflow split is random and causes data leakage.

```python
import json
from pathlib import Path
from torch.utils.data import Dataset
from PIL import Image
import numpy as np

CLASSES = [
    "background", "building_damaged", "building_intact", "debris_rubble",
    "fire_smoke", "road_blocked", "road_clear", "vegetation", "vehicle", "water_flood",
]
NUM_CLASSES = len(CLASSES)

def load_split_manifest(manifest_path="split_manifest.json"):
    with open(manifest_path) as f:
        return json.load(f)

def get_split_frames(manifest, split, images_root="exports/disasterview-raw/images",
                     labels_root="exports/disasterview-raw/labels"):
    """Return list of (image_path, label_path) tuples for a given split."""
    split_keys = manifest["splits"][split]
    pairs = []
    for key in split_keys:
        info = manifest["videos"][key]
        dtype, vid_id = info["disaster_type"], info["video_id"]
        img_dir = Path(images_root) / dtype / vid_id
        lbl_dir = Path(labels_root) / dtype / vid_id
        for img_path in sorted(img_dir.glob("*.jpg")):
            lbl_path = lbl_dir / img_path.with_suffix(".txt").name
            if lbl_path.exists():
                pairs.append((img_path, lbl_path))
    return pairs


class DisasterViewDataset(Dataset):
    def __init__(self, frame_pairs, img_size=512, transform=None):
        self.pairs = frame_pairs
        self.img_size = img_size
        self.transform = transform

    def __len__(self):
        return len(self.pairs)

    def yolo_seg_to_mask(self, txt_path, h, w):
        """Convert YOLO-seg .txt to HxW integer class mask."""
        mask = np.zeros((h, w), dtype=np.int64)
        if not Path(txt_path).exists():
            return mask
        import cv2
        for line in Path(txt_path).read_text().strip().splitlines():
            parts = line.strip().split()
            if len(parts) < 5:
                continue
            class_id = int(parts[0])
            coords = list(map(float, parts[1:]))
            pts = np.array(
                [[int(coords[i] * w), int(coords[i+1] * h)]
                 for i in range(0, len(coords)-1, 2)],
                dtype=np.int32
            )
            cv2.fillPoly(mask, [pts], class_id)
        return mask

    def __getitem__(self, idx):
        img_path, lbl_path = self.pairs[idx]
        image = Image.open(img_path).convert("RGB").resize(
            (self.img_size, self.img_size), Image.BILINEAR
        )
        mask = self.yolo_seg_to_mask(lbl_path, self.img_size, self.img_size)
        if self.transform:
            image = self.transform(image)
        return image, mask


# Usage
manifest = load_split_manifest()
train_pairs = get_split_frames(manifest, "train")
val_pairs   = get_split_frames(manifest, "val")
test_pairs  = get_split_frames(manifest, "test")

print(f"Train: {len(train_pairs)}, Val: {len(val_pairs)}, Test: {len(test_pairs)}")
```

## Training SegFormer-B0

```python
from transformers import SegformerForSemanticSegmentation, SegformerImageProcessor
import torch
from torch.utils.data import DataLoader
from torchvision import transforms

# Preprocessing
processor = SegformerImageProcessor(
    do_resize=True, size={"height": 512, "width": 512},
    do_normalize=True,
    image_mean=[0.485, 0.456, 0.406],
    image_std=[0.229, 0.224, 0.225],
)

# Model — SegFormer-B0 pretrained on ADE20K, fine-tuned on DisasterView
model = SegformerForSemanticSegmentation.from_pretrained(
    "nvidia/segformer-b0-finetuned-ade-512-512",
    num_labels=10,
    ignore_mismatched_sizes=True,
    id2label={i: c for i, c in enumerate(CLASSES)},
    label2id={c: i for i, c in enumerate(CLASSES)},
)

# Class frequencies for weighted loss (inverse of area percentage)
class_weights = torch.tensor([
    1/14.4, 1/24.0, 1/3.1, 1/10.8, 1/6.7,
    1/4.5,  1/2.4,  1/4.0, 1/2.3,  1/27.7,
], dtype=torch.float32)
class_weights = class_weights / class_weights.sum() * 10  # normalize

criterion = torch.nn.CrossEntropyLoss(weight=class_weights, ignore_index=255)

optimizer = torch.optim.AdamW(model.parameters(), lr=6e-5, weight_decay=0.01)
scheduler = torch.optim.lr_scheduler.PolynomialLR(optimizer, total_iters=160, power=1.0)

# Training loop (simplified)
device = "cuda" if torch.cuda.is_available() else "cpu"
model = model.to(device)

for epoch in range(160):
    model.train()
    for images, masks in DataLoader(
        DisasterViewDataset(train_pairs), batch_size=8, shuffle=True, num_workers=4
    ):
        inputs = processor(images=list(images), return_tensors="pt").to(device)
        labels = masks.to(device)
        outputs = model(**inputs, labels=labels)
        loss = outputs.loss
        loss.backward()
        optimizer.step()
        optimizer.zero_grad()
    scheduler.step()
    print(f"Epoch {epoch+1}/160, Loss: {loss.item():.4f}")
```

## Evaluation and Per-Class mIoU

```python
import torch
import numpy as np
from tqdm import tqdm

def compute_iou(pred_mask, true_mask, num_classes=10):
    ious = []
    for cls in range(num_classes):
        pred_cls = (pred_mask == cls)
        true_cls = (true_mask == cls)
        intersection = (pred_cls & true_cls).sum().item()
        union = (pred_cls | true_cls).sum().item()
        if union == 0:
            ious.append(float("nan"))
        else:
            ious.append(intersection / union)
    return ious

model.eval()
all_ious = [[] for _ in range(10)]

with torch.no_grad():
    for images, masks in tqdm(DataLoader(
        DisasterViewDataset(test_pairs), batch_size=4, num_workers=4
    )):
        inputs = processor(images=list(images), return_tensors="pt").to(device)
        outputs = model(**inputs)
        logits = outputs.logits  # (B, C, H/4, W/4)
        preds = torch.nn.functional.interpolate(
            logits, size=(512, 512), mode="bilinear", align_corners=False
        ).argmax(dim=1).cpu()

        for pred, true in zip(preds, masks):
            frame_ious = compute_iou(pred, true)
            for cls, iou in enumerate(frame_ious):
                if not np.isnan(iou):
                    all_ious[cls].append(iou)

print("\nPer-class IoU on test split (video-disjoint):")
print(f"{'Class':<22} {'IoU':>6}")
print("-" * 30)
miou_vals = []
for cls, ious in enumerate(all_ious):
    mean_iou = np.mean(ious) if ious else float("nan")
    miou_vals.append(mean_iou)
    print(f"{CLASSES[cls]:<22} {mean_iou:.4f}")
print("-" * 30)
print(f"{'mIoU':<22} {np.nanmean(miou_vals):.4f}")
```
